Валидация формы

JavaScript • 4.7 ★ • 18 скачиваний

Универсальная функция валидации email, пароля и номера телефона. Подходит для любых форм.

JS 
const validateForm = ({ email, password, phone }) => {
  const errors = {};
  
  if (!email.includes('@')) errors.email = 'Invalid email';
  if (password.length < 8) errors.password = 'Too short';
  if (!/^\+?[\d\s-]{10,}$/.test(phone)) {
    errors.phone = 'Invalid phone number';
  }
  
  return {
    isValid: Object.keys(errors).length === 0,
    errors
  };
};
